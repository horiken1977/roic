Jenkinsパイプライン構成

    agent any
    
    environment {
        AWS_REGION = 'ap-northeast-1'
        TOMCAT_URL = 'http://54.199.201.201:8080'
    }
    
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/horiken1977/roic.git'
            }
        }
        
        stage('Build') {
            steps {
                sh 'npm install'
                sh 'npm run build'
            }
        }
        
        stage('Test') {
            steps {
                sh 'npm run test'
                sh 'npm run test:coverage'
            }
        }
        
        stage('Deploy') {
            steps {
                sh './scripts/deploy.sh'
            }
        }
    }
}
